#include <iostream>
using namespace std;

#define MAX 10

class Queue {
public:
    int a[MAX];
    int front, rear;

    Queue() {
        front = rear = -1;
    }

    bool isFull() {
        return rear == MAX - 1;
    }

    bool isEmpty() {
        return front == -1;
    }

    void enqueue(int x) {
        if (isFull()) return;
        if (front == -1) front = 0;
        a[++rear] = x;
    }

    void dequeue() {
        if (isEmpty()) return;
        if (front == rear) front = rear = -1;
        else front++;
    }

    void display() {
        if (isEmpty()) return;
        for (int i = front; i <= rear; i++)
            cout << a[i] << " ";
        cout << endl;
    }
};

int main() {
    Queue q;
    int ch, x;

    do {
        cin >> ch;

        if (ch == 1) {
            cin >> x;
            q.enqueue(x);
        }
        else if (ch == 2) {
            q.dequeue();
        }
        else if (ch == 3) {
            q.display();
        }

    } while (ch != 4);

    return 0;
}
